class Solution {
public:
    ListNode* FindKthToTail(ListNode* pListHead, unsigned int k) {
        // 分为两段，第一段先移动K，后面随之移动 只需遍历一遍
        ListNode *p //= (struct ListNode *)malloc(sizeof(struct ListNode));
        p = pListHead;
        for (int i = 0; i < k; i++)
        {
            if (p == NULL)
                return NULL;
            p = p->next;
        }
        while (p != NULL)
        {
            pListHead = pListHead->next;
            p = p->next;
        }
        return pListHead;
    }
};
